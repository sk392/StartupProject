package com.sk392.kr.carmony.Main; 
 
import android.content.Context; 
import android.os.Bundle; 
import android.support.design.widget.AppBarLayout; 
import android.support.design.widget.TabLayout; 
import android.support.v7.app.AppCompatActivity; 
import android.support.v7.widget.Toolbar; 
import android.view.View; 
import android.widget.Button; 
import android.widget.ImageButton; 
import android.widget.TextView; 
import android.widget.Toast; 
 
import com.sk392.kr.carmony.Library.BackPressClosehHandler;
import com.sk392.kr.carmony.Library.NonSwipeViewPager; 
import com.sk392.kr.carmony.R;

public class MainActivity extends AppCompatActivity { 
    public static final String URL_AUTH = "https://14.63.225.26:27443/api/user_auth.php"; 
    public static final String URL_USER_INFO = "http://14.63.225.26:2780/api/get_user_info.php"; 
    public static final String SHARED_MAIN_NAME="sharedsk!mainname"; 
    public static final String SHARED_TOKEN="sharedsk!token"; 
    public static final String SHARED_EMAIL="sharedsk!email"; 
    public static final String SHARED_NAME="sharedsk!name"; 
    public static final String SHARED_TYPE="sharedsk!type"; 
    public static final String SHARED_USER_ID="sharedsk!userid"; 
    public static final String SHARED_CONTENT="sharedsk!content"; 
    public static final String SHARED_IS_FACE="sharedsk!isface"; 
    public static final String SHARED_USER_REVIEW_CNT="sharedsk!userreviewcnt"; 
    public static final String SHARED_USER_REVIEW_SCORE="sharedsk!userreviewscore"; 
    public static final String SHARED_OWNER_REVIEW_CNT="sharedsk!reviewcnt"; 
    public static final String SHARED_OWNER_REVIEW_SCORE="sharedsk!reviewscore"; 
    public static final String SHARED_IS_EMAIL="sharedsk!isemail"; 
    public static final String SHARED_IS_COMPANY="sharedsk!iscompany"; 
    public static final String SHARED_IS_PHONE="sharedsk!isphone"; 
    public static final String SHARED_IS_LICEN="sharedsk!islicen"; 
    public static final String SHARED_MILEAGE="sharedsk!mileage"; 
    public static final String SHARED_ACCIDENT="sharedsk!accident"; 
    public static final String SHARED_USER_RES_CNT="sharedsk!userrescnt"; 
    public static final String SHARED_USER_PHONE="sharedsk!userphone"; 
    boolean isMain = true; 
    PagerAdapter adapter; 
    NonSwipeViewPager viewPager; 
    TabLayout tabLayout; 
    Toolbar toolbar; 
    OwnerRootFragment ownerRootFragment; 
    ReservationFragment resFragment; 
    RootFragment rootFragment; 
    UserinfoFragment userinfoFragment; 
    ChatListFragment chatListFragment; 
    Button btToolbar; 
    ImageButton ibToolbarUser,ibToolbarOwner; 
    TextView tvTitle; 
    private BackPressClosehHandler backPressClosehHandler; 
    public static String SENDBIRD_APP_ID = "03204093-25A1-491C-8658-D108A9E533B2"; 
 
    @Override 
    protected void onCreate(Bundle savedInstanceState) { 
 
 
        super.onCreate(savedInstanceState); 
        setContentView(R.layout.activity_main); 
        backPressClosehHandler = new BackPressClosehHandler(this); 
 
        final AppBarLayout appBarLayout = (AppBarLayout) findViewById(R.id.appbar); 
 
        toolbar = (Toolbar) findViewById(R.id.introtoolbar); 
        toolbar.setTitle(""); 
        btToolbar = (Button) toolbar.findViewById(R.id.bt_toolbar); 
        ibToolbarUser = (ImageButton) toolbar.findViewById(R.id.ib_toolbar_user); 
        ibToolbarOwner = (ImageButton) toolbar.findViewById(R.id.ib_toolbar_owner); 
        tvTitle = (TextView) toolbar.findViewById(R.id.tv_toolbar); 
        setSupportActionBar(toolbar); 
 
        viewPager = (NonSwipeViewPager) findViewById(R.id.viewpager); 
        if (viewPager != null) { 
            setupViewPager(); 
        } 
 
        tabLayout = (TabLayout) findViewById(R.id.tabs); 
        tabLayout.setupWithViewPager(viewPager); 
        tabLayout.getTabAt(0).setIcon(R.drawable.ic_search_black_24dp); 
        tabLayout.getTabAt(1).setIcon(R.drawable.ic_date_range_black_24dp); 
        tabLayout.getTabAt(2).setIcon(R.drawable.ic_sms_black_24dp); 
        tabLayout.getTabAt(3).setIcon(R.drawable.ic_drive_eta_black_24dp); 
        tabLayout.getTabAt(4).setIcon(R.drawable.ic_person_black_24dp); 
        tvTitle.setText(R.string.search_fage_name); 
 
 
        //탭 화면이 바뀔 때를 인식하는 리스너 
        tabLayout.setOnTabSelectedListener(new TabLayout.OnTabSelectedListener() { 
            @Override 
            public void onTabSelected(TabLayout.Tab tab) { 
                //다른 화면에 넘어가면 앱바를 확장시킨다. 
                appBarLayout.setExpanded(true); 
                //탭을 눌렀을 때 화면도 바꿔준다. 
                viewPager.setCurrentItem(tab.getPosition()); 
                getSupportFragmentManager().popBackStack(); 
                if (tab.getPosition() == 0) { 
                    tvTitle.setText(R.string.search_fage_name); 
                    btToolbar.setVisibility(View.GONE); 
                    ibToolbarUser.setVisibility(View.GONE); 
                    ibToolbarOwner.setVisibility(View.GONE); 
                } else if (tab.getPosition() == 1) { 
                    btToolbar.setVisibility(View.VISIBLE); 
                    ibToolbarUser.setVisibility(View.GONE); 
                    ibToolbarOwner.setVisibility(View.GONE); 
                    btToolbar.setBackgroundResource(R.drawable.round_button_type2); 
                    if (resFragment.getisOwner()) { 
                        tvTitle.setText(R.string.reservation_fage_name_owner); 
                        btToolbar.setText("대여"); 
                    } else { 
                        tvTitle.setText(R.string.reservation_fage_name); 
                        btToolbar.setText("차주"); 
 
                    } 
                } else if (tab.getPosition() == 2) { 
 
                    tvTitle.setText(R.string.message); 
                    btToolbar.setVisibility(View.GONE); 
                    ibToolbarUser.setVisibility(View.GONE); 
                    ibToolbarOwner.setVisibility(View.GONE); 
                } else if (tab.getPosition() == 3) { 
                    //조건문 걸어서 차주등록이 되어있으면 설정창 보이게 아니면 안보이게! 
                    ibToolbarOwner.setBackgroundResource(R.drawable.setting_btn);
 
                    ibToolbarUser.setVisibility(View.GONE); 
                    tvTitle.setText(R.string.owner_fage_name); 
                    btToolbar.setVisibility(View.GONE); 
                } else if (tab.getPosition() == 4) { 
                    btToolbar.setVisibility(View.GONE); 
                    ibToolbarUser.setBackgroundResource(R.drawable.setting_btn);
                    ibToolbarUser.setVisibility(View.VISIBLE); 
                    tvTitle.setText(R.string.profile); 
                    ibToolbarOwner.setVisibility(View.GONE); 
                } 
            } 
 
            @Override 
            public void onTabUnselected(TabLayout.Tab tab) { 
 
            } 
 
            @Override 
            public void onTabReselected(TabLayout.Tab tab) { 
 
            } 
        }); 
    } 
 
    private void setupViewPager() { 
        adapter = new PagerAdapter(getSupportFragmentManager()); 
        ownerRootFragment = new OwnerRootFragment(); 
        resFragment = new ReservationFragment(); 
        rootFragment = new RootFragment(); 
        userinfoFragment = new UserinfoFragment(); 
        chatListFragment = new ChatListFragment(); 
        //각각의 프래그먼트를 어뎁터에 추가해주고 뷰페이저에 해당 어뎁터를 뷰페이저에 매칭한다. 
        adapter.addFragment(rootFragment); 
        adapter.addFragment(resFragment); 
        adapter.addFragment(chatListFragment); 
        adapter.addFragment(ownerRootFragment); 
        adapter.addFragment(userinfoFragment); 
        viewPager.setAdapter(adapter); 
 
    } 
 
    public static void showToast(Context context, String content) { 
        Toast.makeText(context, content, Toast.LENGTH_SHORT).show(); 
    } 
 
    @Override 
    public void onBackPressed() { 
        if (isMain) { 
            backPressClosehHandler.onBackPressed(); 
        } else { 
            super.onBackPressed(); 
        } 
    } 
 
 
    public void setIsMain(boolean isMain) { 
        this.isMain = isMain; 
    } 
}
